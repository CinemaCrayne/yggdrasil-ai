    ],
    ".": [

        "setup.py",
        "requirements.txt",
        ".env"
    ]
}

DEFAULT_FILE_CONTENT = {
    "run_minerva.py": '''from yggdrasil_ai.minerva_loop import minerva_reflection_loop
from dotenv import load_dotenv; load_dotenv()

if __name__ == "__main__":
    print("ğŸŒ³ Starting Minerva loop...")

''',

def ensure_structure(base_path):
    print(f"ğŸ” Checking layout in: {base_path}\n")
    errors = 0

    for folder, files in REQUIRED_STRUCTURE.items():
        folder_path = os.path.join(base_path, folder)
        if not os.path.isdir(folder_path):
            print(f"ğŸ“ Creating missing folder: {folder_path}")
            os.makedirs(folder_path)
        for file in files:
            file_path = os.path.join(folder_path, file)
            if not os.path.isfile(file_path):
                print(f"ğŸ“ Creating: {file_path}")
                with open(file_path, 'w', encoding='utf-8') as f:
                    content_key = f"{folder}/{file}" if folder != "." else file
                    f.write(DEFAULT_FILE_CONTENT.get(content_key, "# Placeholder\n"))
                errors += 1
            else:
                print(f"âœ… Found: {file_path}")

    if errors == 0:
        print("\nâœ… All files and folders are present.")
    else:
        print(f"\nğŸ›  Created {errors} missing file(s). You're ready to go!")

if __name__ == "__main__":
    ensure_structure(os.getcwd())
