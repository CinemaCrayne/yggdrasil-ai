def replace_emojis(line):
    for emoji, replacement in EMOJI_MAP.items():
        if emoji in line:
            line = line.replace(emoji, replacement)
    return line

def sanitize_file(filepath: Path):
    if filepath.suffix != ".py":
        return

    with filepath.open(encoding="utf-8") as f:
        lines = f.readlines()

    changed = False
    new_lines = []

    for line in lines:
        if "print(" in line or "logging." in line:
            new_line = replace_emojis(line)
            if new_line != line:
                changed = True
            new_lines.append(new_line)
        else:
            new_lines.append(line)

    if changed:
        with filepath.open("w", encoding="utf-8") as f:
            f.writelines(new_lines)
        print(f"ğŸ§¼ Cleaned emojis in: {filepath}")

def sanitize_directory(base: Path):
    for path in base.rglob("*.py"):
        sanitize_file(path)
